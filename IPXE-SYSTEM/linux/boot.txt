#!ipxe

menu 
item tinycore384 tinycore384
item tinycore60 tinycore60
item iso iso
item shell shell
choose --default exit --timeout 3000 target && goto ${target}

:iso
kernel http://static.netboot.me/memdisk-iso
initrd http://distro.ibiblio.org/tinycorelinux/3.x/release/tinycore_3.8.4.iso
boot

:tinycore384
kernel tiny384/bzImage quiet ramdisk_size=10000 pause httplist=http://sxoleio.besaba.com/w/linux/tce/extension.lst  bootfile=http://sxoleio.besaba.com/w/linux/ 
initrd tiny384/tinycore.gz
initrd tce/cm.txt /home/tc/cm.txt
initrd tce/cmex /home/tc/cmex
initrd tce/cmwm /home/tc/cmwm
initrd tce/rdpcon /home/tc/rdpcon
initrd tce/rdpwin /home/tc/rdpwin

initrd tce/rdesktop.tcz /home/tc/rdesktop.tcz
initrd tce/libusb.tcz /home/tc/libusb.tcz
initrd tce/glib2.tcz /home/tc/glib2.tcz
initrd tce/libffi.tcz /home/tc/libffi.tcz
initrd tce/libao.tcz /home/tc/libao.tcz
initrd tce/libasound.tcz  /home/tc/libasound.tcz
initrd tce/udev-lib.tcz /home/tc/udev-lib.tcz
initrd tce/pcsc-lite.tcz /home/tc/pcsc-lite.tcz
initrd tce/openssl-1.0.1.tcz /home/tc/openssl-1.0.1.tcz

boot

:shell
shell

:tinycore60
kernel tinycore60/vmlinuz quiet ramdisk_size=10000 pause bootfile=http://sxoleio.besaba.com/w/linux/ 
initrd tinycore60/core.gz
initrd tce/cm.txt /opt/cm.txt
initrd tce/cmex /opt/cmex
initrd tce/cmwm /opt/cmwm
initrd tce/rdpcon /opt/rdpcon
initrd tce/rdpwin /opt/rdpwin

boot
